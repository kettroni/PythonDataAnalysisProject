# Put your solution here!
def rescale(serie):
    serie = pd.Series(serie)
    mean = serie.mean()
    std = serie.std()
    s2 = serie.apply(lambda x: (x-mean)/(2*std))
    return s2

# Put your solution here!
def train_test_split(df, train_fraction=0.8):
    train = df.sample(frac=train_fraction)
    test = df.drop(train.index)
    
    return train,test
